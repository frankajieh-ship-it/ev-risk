import React, { useMemo, useState } from "react";

type RiskLevel = "green" | "yellow" | "red";

type FormState = {
  vinOrModel: string;
  zip: string;
  dailyMiles: string;
  homeCharging: "yes" | "no";
  riskTolerance: "low" | "medium" | "high";
};

type ResultState = {
  level: RiskLevel;
  score: number; // 0â€“100, higher = better (buy confidence)
  verdict: string;
  reasons: string[];
};

function clamp(n: number, min: number, max: number) {
  return Math.max(min, Math.min(max, n));
}

function levelFromScore(score: number): RiskLevel {
  if (score >= 70) return "green";
  if (score >= 45) return "yellow";
  return "red";
}

function levelEmoji(level: RiskLevel) {
  return level === "green" ? "ðŸŸ¢" : level === "yellow" ? "ðŸŸ¡" : "ðŸ”´";
}

function levelLabel(level: RiskLevel) {
  return level === "green" ? "Good Buy" : level === "yellow" ? "Proceed Carefully" : "High Risk";
}

/**
 * MVP placeholder scoring (rules-first).
 * Replace with your real scoring engine when backend is ready.
 * IMPORTANT: This is intentionally simple for UI validation.
 */
function computeMvpResult(form: FormState): ResultState {
  const daily = Number(form.dailyMiles || "0");
  const hasVin = /^[A-HJ-NPR-Z0-9]{11,17}$/i.test(form.vinOrModel.trim());
  const zipOk = /^\d{5}$/.test(form.zip.trim());
  const dailyOk = Number.isFinite(daily) && daily > 0;

  // Base score
  let score = 60;

  // Confidence adjustments
  if (hasVin) score += 5; // VIN tends to mean the user is further in buying intent
  if (!zipOk) score -= 8;
  if (!dailyOk) score -= 10;

  // Usage stress / fit
  if (daily >= 80) score -= 10;
  else if (daily >= 50) score -= 6;

  // Charging dependency
  if (form.homeCharging === "no") score -= 10;

  // Risk tolerance: if user is risk-averse, tighten thresholds (lower effective score)
  if (form.riskTolerance === "low") score -= 6;
  if (form.riskTolerance === "high") score += 4;

  score = clamp(score, 5, 95);
  const level = levelFromScore(score);

  const verdict =
    level === "green"
      ? "This looks like a solid used-EV choice for your daily driving and charging setup."
      : level === "yellow"
        ? "This could work, but you should verify a few key battery and ownership risks before buying."
        : "High risk for your situationâ€”there are multiple red flags to verify before you commit.";

  const reasons: string[] = [];

  // Plain-English reasons (UI demo)
  if (form.homeCharging === "no") reasons.push("No home charging increases cost and inconvenienceâ€”public charging reliability matters more.");
  if (daily >= 80) reasons.push("High daily mileage leaves less buffer for cold/heat range loss and battery wear.");
  if (!hasVin) reasons.push("No VIN providedâ€”model/year/trim uncertainty makes battery and recall risk harder to verify.");
  if (!zipOk) reasons.push("ZIP code missing/invalidâ€”local climate and charger availability can materially change ownership risk.");
  if (reasons.length === 0) reasons.push("Inputs indicate a typical use case with manageable reliability and ownership risk.");

  // Ensure 3 bullets for consistency
  while (reasons.length < 3) reasons.push("Weâ€™ll confirm battery health, recall history, and range fit in the full report.");

  return { level, score, verdict, reasons: reasons.slice(0, 3) };
}
