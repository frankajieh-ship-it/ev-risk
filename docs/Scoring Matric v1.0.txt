def calculate_score(ev_data, user_profile):
    # 1. Battery Risk (40% of total)
    battery_base = chemistry_score[ev_data.chemistry]
    battery_base *= age_mileage_multiplier(ev_data.age, ev_data.mileage)
    battery_base *= climate_stress_multiplier(user_profile.zip_code)
    battery_score = battery_base * 0.40  # Weighted
    
    # 2. Platform Risk (30% of total)
    platform_base = maturity_score[ev_data.model_year]
    platform_base *= recall_multiplier(ev_data.recall_count)
    platform_base -= known_failure_deductions(ev_data.model)
    platform_score = max(0, platform_base) * 0.30  # Weighted
    
    # 3. Ownership Fit (30% of total)
    ownership_base = range_adequacy_score(user_profile.daily_miles, ev_data.real_range)
    ownership_base *= charging_score[user_profile.charging_access]
    ownership_base *= climate_penalty_multiplier(user_profile.climate, ev_data.heat_pump)
    ownership_score = ownership_base * 0.30  # Weighted
    
    # Final Score (0-100)
    total_score = battery_score + platform_score + ownership_score
